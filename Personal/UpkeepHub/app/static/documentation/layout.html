Header

Stilearn Metro using bootstrap navbar as part of the header:

<!-- start header-->
<header class="header">
    <!-- start navbar, this navbar on top -->
    <div class="navbar navbar-inverse">
        <!-- navbar inner-->
        <div class="navbar-inner">
            <!-- container-->
            <div class="container">
                
                <!-- this btn-navbar contains the menu on the side-left, will be seen on portrait tablet and less. -->
                <a class="btn btn-navbar help-inline" data-toggle="collapse" data-target="#side-left">
                    <span class="icon-bar"></span>
                    <span class="icon-bar"></span>
                    <span class="icon-bar"></span>
                </a>
                
                <!-- Your brand here, images or text -->
                <a class="brand" href="index.html">
                    <!-- just a sample brand & logo -->
                    <i class="aweso-th-large"/> <!-- or use this <img src="path/to/your/brand" alt="" /> -->
                    Stilearn
                </a>
                
                <!-- Un-collapse nav -->
                <div class="nav-uncollapse">
                    <!-- pull left menu and hidden on phone device-->
                    <ul class="nav pull-left hidden-phone">
                        
                    </ul><!--/pull left menu-->

                    <!-- pull right menu-->
                    <ul class="nav pull-right">
                        
                    </ul><!--/pull right menu-->
                </div><!-- /uncollapse nav -->
                
                <!-- Everything you want hidden at 940px or less, leave it blank! (this use when side-left collapse) -->
                <div id="side-left" class="nav-collapse collapse hidden-desktop"></div>
                
            </div><!--/container-->
        </div><!--/navbar-inner-->
        
    </div><!--/ navbar-->
</header>
<!--/ end header-->

Section Content

section-content is a container to your application, here you can put side navs and content of the application that you created:

<!-- start section content-->
<section class="section-content">
    <!-- side left, its part to menu on left-->
    <div class="side-left">
        <!-- navs content -->
    </div><!--/ side left-->


    <!-- start content -->
    <div class="content">
        <!-- your awesome apps content -->
    </div><!--/ end content -->

</section>
<!-- /end section content-->

Side Navs

Side Navs is placed on the section class .sections-content, basic template:

<!-- side left, its part to menu on left-->
<div class="side-left">
    <!--nav, this structure create with nav (find the bootstrap doc about .nav list) -->
    <ul class="nav nav-list">
        <li class="active"><a href="#">Dashboard</a></li>
    </ul><!--/ nav -->
</div><!--/ side left-->

Side navs provides 3 options for child menus (dropdown-list, dropdown and dropup). You can use any or all of them.

<!-- side left, its part to menu on left-->
<div id="navside" class="side-left" data-collapse="navbar">
    <form class="form-inline search-module" action="?" method="post">
        <div class="input-append input-append-inline">
            <input name="search" class="input-block-level" type="text" placeholder="Type to search" />
            <button class="btn bg-cyan" type="button">
                <i class="aweso-search"></i>
            </button>
        </div>
    </form>
    
    <!--nav, this structure create with nav (find the bootstrap doc about .nav list) -->
    <ul class="nav nav-list">
        <li class="active"><a href="#"><Dashboard</a></li>
        <li class="dropdown-list"> <!-- example use with drop list -->
            <a href="#" class="dropdown-toggle" data-toggle="dropdown-list">
                Menus #1 <i class="aweso-angle-down"></i>
            </a>
            <ul class="dropdown-menu">
                <li><a href="#">Child menus #1</a></li>
                <li><a href="#">Child menus #2</a></li>
                <li><a href="#">Child menus #3</a></li>
            </ul>
        </li>
        <li class="dropdown"> <!-- example use with dropdown -->
            <a href="#" class="dropdown-toggle" data-toggle="dropdown">
                Menus #2 <i class="aweso-angle-down"></i>
            </a>
            <ul class="dropdown-menu">
                <li><a href="#">Child menus #1</a></li>
                <li><a href="#">Child menus #2</a></li>
                <li><a href="#">Child menus #3</a></li>
            </ul>
        </li>
        <li class="dropup"> <!-- example use with dropup -->
            <a href="#" class="dropdown-toggle" data-toggle="dropdown">
                Menus #3 <i class="aweso-angle-down"></i>
            </a>
            <ul class="dropdown-menu">
                <li><a href="#">Child menus #1</a></li>
                <li><a href="#">Child menus #2</a></li>
                <li><a href="#">Child menus #3</a></li>
            </ul>
        </li>
    </ul><!--/ nav -->
</div><!--/ side left-->

Contents

Basic HTML template would look as follows:

<!-- start content -->
<div class="content">
    <!-- content header -->
    <header class="content-header">
        <!-- content action (optional)-->
        <ul class="content-action pull-right"></ul>
        
        <!-- content title-->
        <div class="page-header"><h1>Dashboard</h1></div>

        <!-- content breadcrumb -->
        <ul class="breadcrumb breadcrumb-inline">
            <li><a href="#">Home</a> <span class="divider"><i class="aweso-angle-right"></i></span></li>
            <li class="active">path</li>
        </ul>
    </header> <!--/ content header -->

    <!-- content page -->
    <article class="content-page">
        <!-- main page -->
        <div class="main-page">
            <div class="content-inner">
                <!-- write your awesome apps here -->
            </div>
        </div>
    </article> <!-- /content page -->

    <!-- Just a simple footer place here. -->
    <footer class="content-footer">
        <p>Copyright &copy; 2013. All Right Reserved.</p>
    </footer><!--/ footer -->

</div><!--/ end content -->
