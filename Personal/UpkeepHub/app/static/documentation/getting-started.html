Overview

Required HTML5 doctype

Stilearn makes use of certain HTML elements and CSS properties that require the use of the HTML5 doctype. Include it at the beginning of all your projects.

<!DOCTYPE html>
<html lang="en">
  ...
</html>

Integration

To use Stilearn Metro, you have to include at least some of the following files:

CSS Required

/* bootstrap css */
<link href="css/bootstrap.css" rel="stylesheet">
<link href="css/bootstrap-responsive.css" rel="stylesheet">

/* stilearn metro css */
<link href="css/metro-bootstrap.css" rel="stylesheet">
<link href="css/metro.css" rel="stylesheet">
<link href="css/metro-responsive.css" rel="stylesheet">
<link href="css/metro-helper.css" rel="stylesheet">

/* add other css file here... */
<link href="css/font-awesome.css" rel="stylesheet">

Javascript Required


<!-- required js -->
<script type="text/javascript" src="http://platform.twitter.com/widgets.js"></script>
<script type="text/javascript" src="js/jquery.js"></script>
<script type="text/javascript" src="js/jquery-ui.js"></script>
<script type="text/javascript" src="js/bootstrap.js"></script>
<!-- optional -->
<script type="text/javascript" src="js/jquery.ui.touch-punch.min.js"></script>

<!-- required -->
<script type="text/javascript" src="js/stilearn-base.js"></script>

<!-- include other js here...  -->

HTML Template

Basic HTML template would look as follows:

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8">
    <title>Stilearn Metro Admin Bootstrap</title>
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta name="description" content="">
    <meta name="author" content="stilearning">
    

    /* bootstrap css */
    <link href="css/bootstrap.css" rel="stylesheet">
    <link href="css/bootstrap-responsive.css" rel="stylesheet">

    /* stilearn metro css */
    <link href="css/metro-bootstrap.css" rel="stylesheet">
    <link href="css/metro.css" rel="stylesheet">
    <link href="css/metro-responsive.css" rel="stylesheet">
    <link href="css/metro-helper.css" rel="stylesheet">

    /* add other css file here... */
    <link href="css/font-awesome.css" rel="stylesheet">

  </head>

  <body>
    <!-- application template would be place here -->
    ...
    
    <!-- required js -->
    <script type="text/javascript" src="http://platform.twitter.com/widgets.js"></script>
    <script type="text/javascript" src="js/jquery.js"></script>
    <script type="text/javascript" src="js/jquery-ui.js"></script>
    <script type="text/javascript" src="js/bootstrap.js"></script>
    <!-- optional -->
    <script type="text/javascript" src="js/jquery.ui.touch-punch.min.js"></script>

    <!-- required -->
    <script type="text/javascript" src="js/stilearn-base.js"></script>

    <!-- include other js here...  -->
  </body>

</html>
